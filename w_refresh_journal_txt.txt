public subroutine f_re_dp ();dw_trace.accepttext()
if dw_trace.getrow() <=0 then return
string ls_reff,ls_curr,ls_voucher,ls_ket,ls_link,ls_voucher_refresh,ls_coa_jaminan,ls_coa_voucher,ls_coa_debet
decimal ldec_kurs,ldec_debet,ldec_debet_kurs,ldec_faktur
string ls_coa_dp,ls_coa_piutang,ls_curr_faktur,ls_cust,ls_order_client
datetime ldt_tgl
long ll_kas,i,ll_rowcount
 ll_rowcount = dw_trace.rowcount()

for i = 1 to ll_rowcount
	st_3.text = 'Checking row : '+string(i)+' of '+string(ll_rowcount)
	ls_link = dw_trace.getitemstring(i,'voucher')
	ls_reff = dw_trace.getitemstring(i,'account_id')
	ls_curr = dw_trace.getitemstring(i,'curr_id')
	ls_voucher = dw_trace.getitemstring(i,'voucher_manual')
	ls_ket = dw_trace.getitemstring(i,'ket')
	ldec_kurs = dw_trace.getitemdecimal(i,'curr_rate')
	ldec_debet = dw_trace.getitemdecimal(i,'debet')
	ldec_debet_kurs = dw_trace.getitemdecimal(i,'debet_kurs')
	ldt_tgl = dw_trace.object.tgl[i]
	
	if isnull(ls_ket) then ls_ket = ''
	
	//Apakah DP ini untuk jaminan? jika jaminan maka COA nya bukan coa DP
	//////Cek Coa DP dari Voucher
	//edit lagi, dari pilihan
	/*
	select account_id
	into :ls_coa_voucher
	from gl_journal
	where voucher_manual = :ls_voucher and urut = 2
	using sqlca;
	*/
	ls_coa_voucher = dw_trace.getitemstring(i,'coa_kredit')
	
	select acc_lebih_bayar,acc_ar
	into :ls_coa_dp,:ls_coa_piutang
	from gl_setup
	using sqlca;
	
	//228-001 - Penerimaan DImuka - Uang Muka Penjualan
	
	if ls_coa_voucher = '' or isnull(ls_coa_voucher) then
		ls_coa_voucher = ls_coa_dp
	end if
	
	//Coa update pake DP atau jaminan??
	/* ambil berdasarkan inputan di form DP, DONAL 20220822
	if ls_coa_voucher = '228-003' then
		ls_coa_debet = ls_coa_voucher
	else
		ls_coa_debet = ls_coa_dp
	end if
	*/
	ls_coa_debet = dw_trace.getitemstring(i,'coa_kredit')
	
	select curr_id,cust_id,order_client,ttl_netto
	into :ls_curr_faktur,:ls_cust,:ls_order_client,:ldec_faktur
	from tsales1
	where bukti_id = :ls_link
	using sqlca;
	if isnull(ls_curr_faktur) then ls_curr_faktur = 'IDR'
	
	if isnull(ls_coa_dp) or ls_coa_dp = '' then
		messagebox('Warning!','COA DP tidak valid...!, Harap setup terlebih dahulu COA DP..',stopsign!)
		return
	end if
	
	dw_bayar1.reset()
	dw_bayar2.reset()
	dw_gl.reset()
	//debet
	dw_gl.setrow(dw_gl.insertrow(0))
	dw_gl.setitem(dw_gl.getrow(),'voucher',ls_reff)
	dw_gl.setitem(dw_gl.getrow(),'voucher_manual',ls_reff)
	dw_gl.setitem(dw_gl.getrow(),'giro_id',ls_voucher)
	dw_gl.setitem(dw_gl.getrow(),'tgl',ldt_tgl)
	dw_gl.setitem(dw_gl.getrow(),'modul_id','DP')
	dw_gl.setitem(dw_gl.getrow(),'site_id',gs_site)
	dw_gl.setitem(dw_gl.getrow(),'urut',dw_gl.getrow())
	dw_gl.setitem(dw_gl.getrow(),'debet',ldec_debet)
	dw_gl.setitem(dw_gl.getrow(),'kredit',0)
	dw_gl.setitem(dw_gl.getrow(),'debet_kurs',ldec_debet_kurs)
	dw_gl.setitem(dw_gl.getrow(),'kredit_kurs',0)
	dw_gl.setitem(dw_gl.getrow(),'doc_reff',ls_order_client)
	//dw_gl.setitem(dw_gl.getrow(),'account_id',ls_coa_dp)
	dw_gl.setitem(dw_gl.getrow(),'account_id',ls_coa_debet)
	dw_gl.setitem(dw_gl.getrow(),'ket','DP '+ls_voucher+' ('+ls_ket+')')
	//kredit
	dw_gl.setrow(dw_gl.insertrow(0))
	dw_gl.setitem(dw_gl.getrow(),'voucher',ls_reff)
	dw_gl.setitem(dw_gl.getrow(),'voucher_manual',ls_reff)
	dw_gl.setitem(dw_gl.getrow(),'giro_id',ls_voucher)
	dw_gl.setitem(dw_gl.getrow(),'tgl',ldt_tgl)
	dw_gl.setitem(dw_gl.getrow(),'modul_id','DP')
	dw_gl.setitem(dw_gl.getrow(),'site_id',gs_site)
	dw_gl.setitem(dw_gl.getrow(),'urut',dw_gl.getrow())
	dw_gl.setitem(dw_gl.getrow(),'debet',0)
	dw_gl.setitem(dw_gl.getrow(),'kredit',ldec_debet)
	dw_gl.setitem(dw_gl.getrow(),'debet_kurs',0)
	dw_gl.setitem(dw_gl.getrow(),'kredit_kurs',ldec_debet_kurs)
	dw_gl.setitem(dw_gl.getrow(),'doc_reff',ls_order_client)
	dw_gl.setitem(dw_gl.getrow(),'account_id',ls_coa_piutang)
	dw_gl.setitem(dw_gl.getrow(),'ket','DP '+ls_voucher+' ('+ls_ket+')')
	
	//////PEMBAYARAN
	//DP application has NO bank - must not copy kas_id from original receipt
	ll_kas = 0
	
	
	dw_bayar1.reset()
	dw_bayar2.reset()
	
	dw_bayar1.setrow(dw_bayar1.insertrow(0))
	dw_bayar1.setitem(dw_bayar1.getrow(),'voucher',ls_reff)
	dw_bayar1.setitem(dw_bayar1.getrow(),'voucher_manual',ls_reff)
	dw_bayar1.setitem(dw_bayar1.getrow(),'flag_bayar',1)
	dw_bayar1.setitem(dw_bayar1.getrow(),'tgl',ldt_tgl)
	dw_bayar1.setitem(dw_bayar1.getrow(),'vendor_id',ls_cust)
	dw_bayar1.setitem(dw_bayar1.getrow(),'site_id',gs_site)
	dw_bayar1.setitem(dw_bayar1.getrow(),'kas_id',ll_kas)
	dw_bayar1.setitem(dw_bayar1.getrow(),'kolektor_id','K001')
	dw_bayar1.setitem(dw_bayar1.getrow(),'curr_id',ls_curr)
	dw_bayar1.setitem(dw_bayar1.getrow(),'kurs',ldec_kurs)
	dw_bayar1.setitem(dw_bayar1.getrow(),'gl_reff',ls_reff)
	dw_bayar1.setitem(dw_bayar1.getrow(),'keterangan','Bayar dari DP '+ls_voucher)
	dw_bayar1.setitem(dw_bayar1.getrow(),'user_id',gs_user)
	
	dw_bayar2.setrow(dw_bayar2.insertrow(0))
	dw_bayar2.setitem(dw_bayar2.getrow(),'voucher',ls_reff)
	dw_bayar2.setitem(dw_bayar2.getrow(),'urut',1)
	dw_bayar2.setitem(dw_bayar2.getrow(),'flag_order',1)
	dw_bayar2.setitem(dw_bayar2.getrow(),'bukti_id',ls_order_client)
	dw_bayar2.setitem(dw_bayar2.getrow(),'nilai_asli',ldec_faktur)
	if ls_curr <> 'IDR' then
		dw_bayar2.setitem(dw_bayar2.getrow(),'nilai_bayar',ldec_debet_kurs)
		dw_bayar2.setitem(dw_bayar2.getrow(),'nilai_bayar_idr',ldec_debet)
	else
		dw_bayar2.setitem(dw_bayar2.getrow(),'nilai_bayar',ldec_debet)
		dw_bayar2.setitem(dw_bayar2.getrow(),'nilai_bayar_idr',ldec_debet_kurs)
	end if
	dw_bayar2.setitem(dw_bayar2.getrow(),'ket',ls_ket)
	dw_bayar2.setitem(dw_bayar2.getrow(),'acc_bayar',ls_coa_piutang)
	//Delete OLD n Update
	dw_gl.accepttext()
	dw_bayar1.accepttext()
	dw_bayar2.accepttext()
	ls_voucher_refresh = ls_reff+'R'
	
	delete from gl_journal
	where voucher = :ls_reff
	using sqlca;
	commit;
	
	delete from tbyr2
	where voucher = :ls_reff
	using sqlca;
	commit;
	delete from tbyr2
	where voucher = :ls_voucher_refresh
	using sqlca;
	commit;
	
	delete from tbyr1
	where voucher = :ls_reff
	using sqlca;
	commit;
	delete from tbyr1
	where voucher = :ls_voucher_refresh
	using sqlca;
	commit;
	
	ls_voucher_refresh = ls_reff+'P'
	delete from tbyr2
	where voucher = :ls_voucher_refresh
	using sqlca;
	commit;
	delete from tbyr1
	where voucher = :ls_voucher_refresh
	using sqlca;
	commit;
	
	dw_gl.update()
	commit;
	dw_bayar1.update()
	commit;
	dw_bayar2.update()
	commit;
next



end subroutine

public subroutine f_re_dp_beli (); decimal ldec_kurs,ldec_kurs_dp
string ls_curr,ls_curr_dp
decimal ldec_dp,ldec_faktur,ldec_dp_kurs
long ll_count,ll_max,ll_kas,i
string ls_periode,ls_max,ls_voucher,ls_reff,ls_kode,ls_vendor,ls_voucher_refresh
string ls_coa_dp,ls_coa_hutang,ls_voucher_dp,ls_bukti_reff,ls_order_client
datetime ldt_tgl
string ls_gl_link,ls_coa_debet

dw_trace.accepttext()

for i = 1 to dw_trace.rowcount()
	
	ldec_dp = dw_trace.getitemdecimal(i,'debet')
	ldec_dp_kurs = dw_trace.getitemdecimal(i,'debet_kurs')
	
	ls_order_client = dw_trace.getitemstring(i,'voucher')
	
	select order_client
	into :ls_reff
	from tstok1
	where bukti_id = :ls_order_client
	using sqlca;
	
	ldt_tgl = dw_trace.object.tgl[i]
	ls_voucher_dp = dw_trace.getitemstring(i,'voucher_manual')
	ls_bukti_reff = dw_trace.getitemstring(i,'ket')
	ls_curr = dw_trace.getitemstring(i,'curr_id')
	ldec_kurs = dw_trace.getitemdecimal(i,'curr_rate')
	ls_gl_link = dw_trace.getitemstring(i,'account_id')
	ls_coa_debet = dw_trace.object.coa_kredit[i]
	
	if isnull(ldec_dp) then ldec_dp = 0
	if isnull(ldec_kurs) then ldec_kurs = 1
	ls_periode = gs_site+string(ldt_tgl,'yymm')
	ls_kode = ls_periode+'DX' //kode DP pembelian
	
	if isnull(ls_coa_debet) then ls_coa_debet = ''
	
	dw_gl.reset()
	
		//Cek coa DP/HUTANG
		select acc_ap,acc_kurang_bayar
		into :ls_coa_hutang,:ls_coa_dp
		from gl_setup
		using sqlca;
		if isnull(ls_coa_hutang) or ls_coa_hutang = '' then
			messagebox('Warning!','Harap setup terlebih dahulu COA Hutang Dagang pada Setup Company..!',stopsign!)
			return
		end if
		if isnull(ls_coa_dp) or ls_coa_dp = '' then
			messagebox('Warning!','Harap setup terlebih dahulu COA DP Pembelian pada Setup Company..!',stopsign!)
			return
		end if
	
		select curr_id,rate_rp,kas_id
		into :ls_curr_dp,:ldec_kurs_dp,:ll_kas
		from gl_journal
		where voucher_manual = :ls_voucher_dp and
		urut = 1
		using sqlca;
		
		//messagebox('',string(ldec_kurs_dp,'#,##0.00'))
		
		if ls_coa_debet = '' then
			ls_coa_debet = ls_coa_dp
		end if
		//debet
		//ls_coa_hutang = dw_trace.getitemstring(i,'coa_kredit')
		dw_gl.setrow(dw_gl.insertrow(0))
		dw_gl.setitem(dw_gl.getrow(),'voucher',ls_gl_link)
		dw_gl.setitem(dw_gl.getrow(),'voucher_manual',ls_gl_link)
		dw_gl.setitem(dw_gl.getrow(),'giro_id',ls_voucher_dp)
		dw_gl.setitem(dw_gl.getrow(),'site_id',gs_site)
		dw_gl.setitem(dw_gl.getrow(),'urut',dw_gl.getrow())
		dw_gl.setitem(dw_gl.getrow(),'tgl',ldt_tgl)
		dw_gl.setitem(dw_gl.getrow(),'modul_id','DP')
		dw_gl.setitem(dw_gl.getrow(),'account_id',ls_coa_hutang)
		//dw_gl.setitem(dw_gl.getrow(),'account_id',ls_coa_debet)
		dw_gl.setitem(dw_gl.getrow(),'debet',ldec_dp)//*ldec_kurs)
		dw_gl.setitem(dw_gl.getrow(),'kredit',0)
		dw_gl.setitem(dw_gl.getrow(),'debet_kurs',ldec_dp_kurs)
		dw_gl.setitem(dw_gl.getrow(),'kredit_kurs',0)
		dw_gl.setitem(dw_gl.getrow(),'ket','Bayar Hutang dari DP '+left(ls_voucher_dp+' ('+ls_bukti_reff+')',100))
		dw_gl.setitem(dw_gl.getrow(),'order_reff',ls_reff)
		dw_gl.setitem(dw_gl.getrow(),'curr_id',ls_curr_dp)
		dw_gl.setitem(dw_gl.getrow(),'rate_rp',ldec_kurs_dp)
		//kredit
		dw_gl.setrow(dw_gl.insertrow(0))
		dw_gl.setitem(dw_gl.getrow(),'voucher',ls_gl_link)
		dw_gl.setitem(dw_gl.getrow(),'voucher_manual',ls_gl_link)
		dw_gl.setitem(dw_gl.getrow(),'giro_id',ls_voucher_dp)
		dw_gl.setitem(dw_gl.getrow(),'site_id',gs_site)
		dw_gl.setitem(dw_gl.getrow(),'urut',dw_gl.getrow())
		dw_gl.setitem(dw_gl.getrow(),'tgl',ldt_tgl)
		dw_gl.setitem(dw_gl.getrow(),'modul_id','DP')
		//dw_gl.setitem(dw_gl.getrow(),'account_id',ls_coa_hutang)
		dw_gl.setitem(dw_gl.getrow(),'account_id',ls_coa_debet)
		dw_gl.setitem(dw_gl.getrow(),'debet',0)
		dw_gl.setitem(dw_gl.getrow(),'kredit',ldec_dp)
		dw_gl.setitem(dw_gl.getrow(),'debet_kurs',0)
		dw_gl.setitem(dw_gl.getrow(),'kredit_kurs',ldec_dp_kurs)
		dw_gl.setitem(dw_gl.getrow(),'ket','Bayar Hutang dari DP '+ls_voucher_dp+' ('+ls_bukti_reff+')')
		dw_gl.setitem(dw_gl.getrow(),'order_reff',ls_reff)
		dw_gl.setitem(dw_gl.getrow(),'curr_id',ls_curr_dp)
		dw_gl.setitem(dw_gl.getrow(),'rate_rp',ldec_kurs_dp)
		
		//PEMBAYARAN
		dw_bayar1.reset()
		dw_bayar2.reset()
		dw_bayar1.insertrow(0)
		dw_bayar1.setitem(1,'tgl',ldt_tgl)
		dw_bayar1.setitem(1,'voucher',ls_gl_link)
		dw_bayar1.setitem(1,'voucher_manual',ls_gl_link)
		dw_bayar1.setitem(1,'site_id',gs_site)
		dw_bayar1.setitem(1,'flag_bayar',2)
		dw_bayar1.setitem(1,'vendor_id',ls_vendor)
		dw_bayar1.setitem(1,'site_id',gs_site)
		dw_bayar1.setitem(1,'kas_id',ll_kas)
		dw_bayar1.setitem(1,'curr_id',ls_curr_dp)
		dw_bayar1.setitem(1,'kurs',ldec_kurs_dp)
		dw_bayar1.setitem(1,'keterangan','Potong Hutang dari DP '+ls_voucher_dp)
		dw_bayar1.setitem(1,'user_id',gs_user)
		//Detail
		dw_bayar2.insertrow(0)
		dw_bayar2.setitem(1,'voucher',ls_gl_link)
		dw_bayar2.setitem(1,'urut',1)
		dw_bayar2.setitem(1,'bukti_id',ls_reff)
		dw_bayar2.setitem(1,'flag_order',2)
		dw_bayar2.setitem(1,'nilai_asli',ldec_faktur)
		dw_bayar2.setitem(1,'nilai_bayar_idr',ldec_dp)
		if ls_curr = 'IDR' then
			dw_bayar2.setitem(1,'nilai_bayar',ldec_dp)
		else
			dw_bayar2.setitem(1,'nilai_bayar',ldec_dp_kurs)
		end if
		dw_bayar2.setitem(1,'acc_bayar',ls_coa_hutang)
		dw_bayar2.setitem(1,'ket',ls_bukti_reff)
		//update
		dw_gl.accepttext()
		dw_bayar1.accepttext()
		dw_bayar2.accepttext()
		
		ls_voucher_refresh = ls_gl_link+'P'
		
		delete from gl_journal
		where voucher = :ls_gl_link
		using sqlca;
		commit;
		delete from tbyr2
		where voucher = :ls_gl_link
		using sqlca;
		commit;
		delete from tbyr2
		where voucher = :ls_voucher_refresh
		using sqlca;
		commit;
		
		delete from tbyr1
		where voucher = :ls_gl_link
		using sqlca;
		commit;
		delete from tbyr1
		where voucher = :ls_voucher_refresh
		using sqlca;
		commit;
		
		dw_gl.update()
		commit;
		dw_bayar1.update()
		commit;
		dw_bayar2.update()
		commit;
	next



end subroutine

	else
		ldec_bayar = ldec_kredit
		ldec_bayar_kurs = ldec_kredit_kurs
	end if
	
	if ls_curr_jual = 'IDR' then
		ldec_set_bayar = ldec_bayar
	else
		//Nilai bayar harus dalam valuta asing (mis. USD untuk invoice USD)
		//Jika debet_kurs < debet maka debet_kurs sudah berisi nilai foreign (mis. 26720 USD)
		//Jika debet_kurs = debet (IDR) atau 0, hitung dari IDR / kurs sebagai fallback
		if ldec_bayar_kurs > 0 and ldec_bayar_kurs < ldec_bayar then
			ldec_set_bayar = ldec_bayar_kurs
		elseif ldec_kurs > 1 then
			ldec_set_bayar = ldec_bayar / ldec_kurs
		else
			ldec_set_bayar = ldec_bayar
		end if
	end if

	dw_sync2.setitem(dw_sync2.getrow(),'nilai_bayar',ldec_set_bayar)
	dw_sync2.setitem(dw_sync2.getrow(),'ket',ls_ket)
	dw_sync2.setitem(dw_sync2.getrow(),'nilai_bayar_idr',ldec_bayar)
	dw_sync2.setitem(dw_sync2.getrow(),'acc_bayar',ls_coa)

//2. Update HPP Average ke tsales2 berdasarkan formula weighted average (SINV + transaksi periode)
//   Formula: HPP_avg = (AWAL_RP + BELI_RP + MUTASI_IN_RP + RET_JUAL_RP - RET_BELI_RP) /
//                      (AWAL    + BELI    + MUTASI_IN    + RET_JUAL    - RET_BELI)
st_1.text = 'Update HPP Average...'
st_2.text = 'Menghitung & update HPP rata-rata ke tsales2...'
UPDATE tsales2
SET hpp = (
   SELECT
      CASE
         WHEN ((ISNULL(aw.awal,0) + ISNULL(beli.beli,0) + ISNULL(beli.mutasi_in,0) + ISNULL(rj.ret_jual,0)) - ISNULL(beli.ret_beli,0)) <> 0
         THEN ((ISNULL(aw.awal_rp,0) + ISNULL(beli_rp.beli,0) + ISNULL(beli_rp.mutasi_in,0) + ISNULL(beli_rp.ekspedisi,0) + ISNULL(rj.ret_jual_rp,0)) - ISNULL(beli_rp.ret_beli,0)) /
              ((ISNULL(aw.awal,0)    + ISNULL(beli.beli,0)    + ISNULL(beli.mutasi_in,0)    + ISNULL(rj.ret_jual,0))    - ISNULL(beli.ret_beli,0))
         ELSE 0
      END
   FROM im_produk p
   LEFT OUTER JOIN (
      SELECT stok_id, SUM(qty) AS awal, SUM(nilai) AS awal_rp
      FROM sinv
      WHERE MONTH(periode) = MONTH(:ldt1) AND YEAR(periode) = YEAR(:ldt1)
        AND stok_id = tsales2.stok_id
      GROUP BY stok_id
   ) aw ON p.produk_id = aw.stok_id
   LEFT OUTER JOIN (
      SELECT t2.stok_id,
         SUM(CASE WHEN t1.tipe_trans = '02' THEN t2.qty ELSE 0 END) AS beli,
         SUM(CASE WHEN t1.tipe_trans = '12' THEN t2.qty ELSE 0 END) AS ret_beli,
         SUM(CASE WHEN t1.tipe_trans = '09' THEN t2.qty ELSE 0 END) AS mutasi_in
      FROM tstok1 t1, tstok2 t2
      WHERE t1.bukti_id = t2.bukti_id
        AND t1.tgl BETWEEN :ldt1 AND :ldt2
        AND ISNULL(t1.order_oke,'N') = 'Y'
        AND ISNULL(t2.qty,0) <> 0
        AND t2.stok_id = tsales2.stok_id
      GROUP BY t2.stok_id
   ) beli ON p.produk_id = beli.stok_id
   LEFT OUTER JOIN (
      SELECT t2.stok_id,
         SUM(CASE WHEN t1.tipe_trans = '02' THEN (t2.netto * ISNULL(t1.kurs,1)) ELSE 0 END) AS beli,
         SUM(CASE WHEN t1.tipe_trans = '12' THEN ABS(t2.netto_hpp)               ELSE 0 END) AS ret_beli,
         SUM(CASE WHEN t1.tipe_trans = '09' THEN t2.netto                         ELSE 0 END) AS mutasi_in,
         SUM(CASE WHEN t1.tipe_trans = '05' THEN ABS(t2.biaya_ekspedisi) * ABS(ISNULL(t2.qty,0)) ELSE 0 END) AS ekspedisi
      FROM tstok1 t1, tstok2 t2
      WHERE t1.bukti_id = t2.bukti_id
        AND t1.tgl BETWEEN :ldt1 AND :ldt2
        AND ISNULL(t1.order_oke,'N') = 'Y'
        AND ISNULL(t2.qty,0) <> 0
        AND t2.stok_id = tsales2.stok_id
      GROUP BY t2.stok_id
   ) beli_rp ON p.produk_id = beli_rp.stok_id
   LEFT OUTER JOIN (
      SELECT t2.stok_id,
         SUM(CASE WHEN t1.tipe_trans IN ('32','26','36') THEN t2.qty ELSE 0 END) AS ret_jual,
         SUM(CASE WHEN t1.tipe_trans IN ('32','26','36') THEN ABS(t2.netto * ISNULL(t1.kurs,1)) ELSE 0 END) AS ret_jual_rp
      FROM tsales1 t1, tsales2 t2
      WHERE t1.bukti_id = t2.bukti_id
        AND t1.tgl BETWEEN :ldt1 AND :ldt2
        AND t1.order_oke = 'Y'
        AND t1.tipe_trans IN ('32','26','36')
        AND ISNULL(t2.qty,0) <> 0
        AND t2.stok_id = tsales2.stok_id
      GROUP BY t2.stok_id
   ) rj ON p.produk_id = rj.stok_id
   WHERE p.stok_item = 'Y' AND p.produk_id = tsales2.stok_id
)
WHERE tsales2.bukti_id IN (
   SELECT bukti_id FROM tsales1
   WHERE tgl BETWEEN :ldt1 AND :ldt2
     AND order_oke = 'Y'
     AND tipe_trans IN ('22','32','26','36','88')
)
USING sqlca;
