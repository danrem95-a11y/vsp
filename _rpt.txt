event dw_1::itemchanged;call super::itemchanged;string ls_vendor,ls_nama

this.accepttext()
if dwo.name = 'vendor' then
	ls_vendor = this.object.vendor[1]
	
   SELECT NAMA
    INTO :ls_nama 
    FROM MCSTSUPP  
    WHERE FLAG_VENDOR = 1 AND VENDOR_ID = :ls_vendor ;
	 
	this.setitem(1,'vendor_desc',ls_nama)
	dw_2.reset()
	dw_2.setfocus()
end if

if dwo.name = 'group_id' then
	ls_vendor = this.object.group_id[1]
	
	SELECT NAMA_group
	INTO  :ls_nama
	FROM im_product_group
	WHERE kode_group = :ls_vendor 	;
	
	this.setitem(1,'group_desc',ls_nama)
	this.triggerevent('ue_retrieve')
	
End If

if dwo.name = 'gudang_id' then
	ls_vendor = this.object.gudang_id[1]
	
	 SELECT NAMA
    INTO :ls_nama 
    FROM MCSTSUPP  
    WHERE FLAG_VENDOR = 3 AND VENDOR_ID = :ls_vendor ;
	
	this.setitem(1,'gudang_desc',ls_nama)
	

	dw_2.reset()
	dw_2.setfocus()
End If


end event

event dw_1::editchanged;call super::editchanged;string ls_nama
if dwo.name = 'cari' then
	this.accepttext()
	ls_nama = upper(this.object.cari[1])
	if isnull(ls_nama) then ls_nama = ''
ls_nama = upper(ls_nama)
	if ls_nama = '' then
		dw_2.setfilter('')
	else	
		dw_2.setfilter('upper(is_find) like "%' + ls_nama + '%"')
	end if
	dw_2.filter()
	dw_2.sort()
	dw_2.groupcalc()
	
end if
end event
