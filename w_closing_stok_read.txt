event buttonclicked;call super::buttonclicked;datetime ldt_periode,ldt_tgl1,ldt_tgl2,ldt_next
long ll_row,i,ll_found,ll_minus,ll_replace,ll_konfirm,ll_count
decimal ld_qty,ld_awal_rp,ld_nilai,ld_hpp,ld_qty_current,ld_qty_awal,ld_qty_update,ld_nilai_awal,ld_nilai_update,ld_nilai_set
decimal ldec_beli,ldec_beli_rp,ldec_mutasi,ldec_mutasi_rp
string ls_stok,ls_group
decimal ldec_debug,ldec_final,ldec_total_qty,ldec_total_nilai
boolean lb_replace = false
if dwo.name = 'b_proses' then
	this.accepttext( )
     ldt_tgl1 = this.object.periode[1]
	ldt_tgl1 = f_bom(ldt_tgl1)
     ldt_tgl2 = f_eom(ldt_tgl1)
	ll_minus = this.object.oto_minus[1]
	ll_replace = this.object.replace_stok[1]
	
	idt_tgl1 = ldt_tgl1
	idt_tgl2 = ldt_tgl2
	// View data first
	dw_view.settransobject(sqlca)
	dw_view.reset()
	ll_row = dw_view.retrieve(ldt_tgl1,ldt_tgl2) //Pindah saldo
	dw_view.accepttext()
	dw_sinv_update.retrieve(ldt_tgl1) //Untuk update current HPP
	
	select dateadd(day,1,:ldt_tgl2)
	into :ldt_next
	from gl_setup
	using sqlca;
	ldt_next = datetime(date(ldt_next))
	
	if Messagebox('Konfirmasi!','Yakin untuk melakukan closing stock ?',Question!,Yesno!,2)=2 then return
	
	//Konfirm jika ini akhir tahun, cek apakah saldo awal tahun berikutnya di replace??
	if string(ldt_next,'mm') = '01'  then
		if ll_replace = 0 then
			ll_konfirm = messagebox("Konfirmasi!","Ini merupakan Closing Akhir Tahun..!~rYakin untuk menimpa saldo awal tahun berikutnya?",question!,yesnocancel!)
			choose case ll_konfirm
				case 3
					return
				case 1
					if messagebox("Konfirmasi!","Anda memilih untuk menimpa saldo awal tahun berikutnya,yakin untuk melanjutkan proses?",question!,yesno!)=2 then return
					lb_replace = true
				case 2
					if messagebox("Konfirmasi!","Anda memilih untuk tidak menimpa saldo awal tahun berikutnya,yakin untuk melanjutkan proses?",question!,yesno!)=2 then return
					lb_replace = false
			end choose
		else
			ll_konfirm = messagebox("Konfirmasi!","Ini merupakan Closing Akhir Tahun..!~rYakin untuk tidak menimpa saldo awal tahun berikutnya?",question!,yesnocancel!)
			choose case ll_konfirm
				case 3
					return
				case 1
					if messagebox("Konfirmasi!","Anda memilih untuk tidak menimpa saldo awal tahun berikutnya,yakin untuk melanjutkan proses?",question!,yesno!)=2 then return
					lb_replace = false
				case 2
					if messagebox("Konfirmasi!","Anda memilih untuk  menimpa saldo awal tahun berikutnya,yakin untuk melanjutkan proses?",question!,yesno!)=2 then return
					lb_replace = true
			end choose
		end if
	end if
	
	
	
	
	hpb_2.maxposition = ll_row
	setpointer(Hourglass!)
	dw_sinv.reset()
	dw_update_minus.reset()
	
	for i = 1 to ll_row
			hpb_2.position = i
			st_1.text = string(i)+'/'+string(ll_row)
			ls_stok 		= dw_view.object.produk_id[i]
			ld_qty  		= dw_view.object.akhir[i]
			ld_qty_awal = dw_view.object.awal[i]
			ld_nilai_awal = dw_view.object.awal_rp[i]
			ld_awal_rp = dw_view.object.awal_rp[i]
			ldec_final 		= dw_view.object.akhir_rpxx[i]
			ld_nilai 		= dw_view.object.akhir_rpxx[i]
