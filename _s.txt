event dw_1::itemchanged;call super::itemchanged;string ls_vendor,ls_vendor_id,ls_nama,ls_flag,ls_order
string ls_order_reff,ls_prinsipal,ls_destination,ls_proyek
string ls_kurs,ls_group_id,ls_keterangan,ls_curr_id,ls_curr,ls_null
decimal ld_ppn,ld_kurs,ld_rate
datetime ldt_tgl
if dwo.name = 'vendor_id' then
	this.accepttext()
	ls_vendor = this.object.vendor_id[1]
	if isnull(ls_vendor_id) then 
		this.setitem(1,'vendor_id','')
		this.setitem(1,'nama_vendor','')
		return
     end if
	
	SELECT MCSTSUPP.NAMA   
    INTO :ls_nama
    FROM MCSTSUPP
    WHERE ( MCSTSUPP.FLAG_VENDOR = 2 ) AND
         ( MCSTSUPP.VENDOR_ID = :ls_vendor ) ;
	if ls_nama = '' then 
		this.setitem(1,'vendor_id','')
		this.setitem(1,'nama_vendor','')
		return
   end if		
			  
	this.setitem(1,'flag_vendor',2)		  
	this.setitem(1,'nama_vendor',ls_nama)
end if

if dwo.name = 'prinsipal_id' then
	this.accepttext()
	ls_vendor = this.object.prinsipal_id[1]
	if isnull(ls_vendor) then ls_vendor = ''
	gs_prinsipal = ls_vendor 
	
	if isnull(ls_vendor_id) then 
		this.setitem(1,'prinsipal_id','')
		this.setitem(1,'nama_prinsipal','')
		return
     end if
	  
	SELECT MCSTSUPP.NAMA   
    INTO :ls_nama
    FROM MCSTSUPP
    WHERE ( MCSTSUPP.FLAG_VENDOR = 1 ) AND
         ( MCSTSUPP.VENDOR_ID = :ls_vendor ) ;
	if ls_nama = '' then 
		this.setitem(1,'prinsipal_id','')
		this.setitem(1,'nama_prinsipal','')
		return
   end if	
	
	this.setitem(1,'flag_prinsipal',1)		  
	this.setitem(1,'nama_prinsipal',ls_nama)
end if

if dwo.name = 'order_reff' then
	this.accepttext()
	ls_flag = this.object.flag_btb[row] 
	ls_order_reff = this.object.order_reff[1]	
	
	if ls_flag = 'PO' then
		SELECT MCSTSUPP.NAMA,   
			   TORDER1.ORDER_ID,
			   TORDER1.PPN,
			   TORDER1.VENDOR_ID,
			   TORDER1.PRINSIPAL_ID,
			   TORDER1.DESTINATION_SITE,
			   torder1.proyek,
			   torder1.curr_id,
			   torder1.kurs,
			   torder1.group_id,
				torder1.keterangan	
    INTO  :ls_nama,   
	 		:ls_order,
			:ld_ppn,
			:ls_vendor,
			:ls_prinsipal,
			:ls_destination,
			:ls_proyek,
			:ls_curr,
			:ld_kurs,
			:ls_group_id,
			:ls_keterangan
    FROM MCSTSUPP,
	 		TORDER1
   WHERE ( TORDER1.VENDOR_ID = MCSTSUPP.VENDOR_ID ) AND
			( TORDER1.FLAG_ORDER = 2 ) AND
         ( TORDER1.ORDER_CLIENT = :ls_order_reff );
			
		this.setitem(1,'vendor_id',ls_vendor)		  
		this.setitem(1,'nama_vendor',ls_nama)
	else
		SELECT MCUST.CUST_NAME,   
					TORDER1.ORDER_ID,
					TORDER1.PPN,
					TORDER1.VENDOR_ID,
					TORDER1.PRINSIPAL_ID,
					TORDER1.DESTINATION_SITE,
					torder1.proyek,
					torder1.curr_id,
					torder1.kurs,
					torder1.group_id,
					torder1.keterangan	
		 INTO  :ls_nama,   
				:ls_order,
				:ld_ppn,
				:ls_vendor,
				:ls_prinsipal,
				:ls_destination,
				:ls_proyek,
				:ls_curr,
				:ld_kurs,
				:ls_group_id,
				:ls_keterangan
		 FROM MCUST,
				TORDER1
		WHERE ( TORDER1.VENDOR_ID = MCUST.CUST_ID ) AND
				( TORDER1.FLAG_ORDER = 1 ) AND
				( TORDER1.ORDER_CLIENT = :ls_order_reff );
		ls_prinsipal = 'P001'
		this.setitem(1,'cust_id',ls_vendor)		  
		this.setitem(1,'cust_name',ls_nama)
	end if
			
	if isnull(ls_prinsipal) then ls_prinsipal = ''
	gs_prinsipal = ls_prinsipal 		
			  
	this.setitem(1,'flag_vendor',2)		  	
	this.setitem(1,'ppn',ld_ppn)
	this.setitem(1,'destination_site',ls_destination)
	this.setitem(1,'proyek',ls_proyek)
	this.setitem(1,'curr_id',ls_curr)
	this.setitem(1,'kurs',ld_kurs)
	this.setitem(1,'keterangan',ls_keterangan)
	
	if ls_flag = 'PO' then
		SELECT MCSTSUPP.NAMA 
		 INTO :ls_nama
		 FROM MCSTSUPP
		WHERE ( MCSTSUPP.FLAG_VENDOR = 1 ) AND
				( MCSTSUPP.VENDOR_ID = :ls_prinsipal );
	else
		ls_nama = 'TRADING'
	end if
	
	this.setitem(1,'flag_prinsipal',1)		  
	this.setitem(1,'prinsipal_id',ls_prinsipal)
	this.setitem(1,'nama_prinsipal',ls_nama)
	
	idw_down3.Settransobject(SQLCA)
	idw_down3.reset()
	idw_down3.retrieve(ls_order)
end if	

if dwo.name = 'tgl' or dwo.name = 'curr_id' then
	ldt_tgl = this.object.tgl[1]
	ldt_tgl = datetime(date(ldt_tgl))
	ls_curr_id = this.object.curr_id[1]
	
	if isnull(ls_curr_id) then ls_curr_id = 'IDR'	
	if ls_curr_id <> 'IDR' then
		select rate_rp INTO :ld_rate from gl_rate where curr_id = :ls_curr_id
		and tgl = :ldt_tgl;
	end if
	if isnull(ld_rate) then ld_rate = 1
	if  ls_curr_id = 'IDR' then ld_rate = 1
	this.setitem(1,'kurs',ld_rate)
end if

end event

elseif idw_up1.object.flag_btb[1] = 'SO' then

		Messagebox('Error','Cust ID must be entry ..!')

where vendor_id = :ls_vendor ;

	Messagebox('Error','Prinsipal ID must be entry ..!')
	return

ls_order 		= idw_up1.object.order_reff[1]
is_key   		= idw_up1.object.bukti_id[1]
if ii_poso = 1 then
	if isnull(ls_order) or trim(ls_order) = '' then
		Messagebox('Error','Please fill in Purchase Order ID ...!')
		Return 
	end if
end if

if isnull(is_key) or trim(is_key) = '' then
	ls_periode = gs_site + string(ldt_tgl,'yymm') + '99'
	SELECT MAX(substring(bukti_id,10,5)) INTO :ls_bukti FROM TSTOK1
	 WHERE left(bukti_id,9) = :ls_periode ;

		ls_flag = '99'

    ls_periode = 'NOITEM' + String(ldt_tgl,'yymm')
	 SELECT MAX(substring(order_client,11,5)) INTO :ls_order_client FROM TSTOK1
	 WHERE left(order_client,10) = :ls_periode ;

	idw_up1.modify('order_client.visible = 0')

// ============================================================
// Validasi Balance Jurnal: total DEBET harus = total KREDIT
// ============================================================
decimal ldec_bal_debet, ldec_bal_kredit
ldec_bal_debet  = 0
ldec_bal_kredit = 0

select sum(isnull(netto,0.00))
into :ldec_bal_debet
from tstok2
where bukti_id = :is_key and upper(sat_kecil) = 'DEBET'
using sqlca;

select sum(isnull(netto,0.00))
into :ldec_bal_kredit
from tstok2
where bukti_id = :is_key and upper(sat_kecil) = 'KREDIT'
using sqlca;

if isnull(ldec_bal_debet)  then ldec_bal_debet  = 0
if isnull(ldec_bal_kredit) then ldec_bal_kredit = 0

if ldec_bal_debet <> ldec_bal_kredit then
   messagebox('Warning! Jurnal Tidak Balance', &
      'Total DEBET  : ' + string(ldec_bal_debet,  '#,###,###,##0.00') + '~r~n' + &
      'Total KREDIT : ' + string(ldec_bal_kredit, '#,###,###,##0.00') + '~r~n~r~n' + &
      'Selisih      : ' + string(abs(ldec_bal_debet - ldec_bal_kredit), '#,###,###,##0.00') + '~r~n~r~n' + &
      'Data tidak dapat disimpan sebelum jurnal balance.', &
      stopsign!)
   return
end if
// ============================================================

	if is_flag = 'PO' then

	dw_update.settransobject(sqlca)
	dw_update.retrieve(is_key)

	dw_update.setitem(1,'tgl',idw_up1.object.tgl[1])
	dw_update.setitem(1,'flag_prinsipal',idw_up1.object.flag_prinsipal[1])
	dw_update.setitem(1,'prinsipal_id',idw_up1.object.prinsipal_id[1])
	dw_update.setitem(1,'order_reff',idw_up1.object.order_reff[1])
	dw_update.setitem(1,'bukti_reff',idw_up1.object.bukti_reff[1])
	dw_update.setitem(1,'from_site',idw_up1.object.from_site[1])
	dw_update.setitem(1,'destination_site',idw_up1.object.destination_site[1])
	dw_update.setitem(1,'ttl_kotor',idw_up1.object.ttl_kotor[1])
	dw_update.setitem(1,'ttl_pot',idw_up1.object.ttl_pot[1])
	dw_update.setitem(1,'ttl_netto',idw_up1.object.ttl_netto[1])
	dw_update.setitem(1,'keterangan',idw_up1.object.keterangan[1])
	dw_update.setitem(1,'ppn',idw_up1.object.ppn[1])
	dw_update.setitem(1,'ttl_ppn',idw_up1.object.ttl_ppn[1])
	dw_update.setitem(1,'flag_vendor',idw_up1.object.flag_vendor[1])
	dw_update.setitem(1,'vendor_id',idw_up1.object.vendor_id[1])
	dw_update.setitem(1,'order_client',idw_up1.object.order_client[1])
	dw_update.setitem(1,'order_oke',idw_up1.object.order_oke[1])
	dw_update.setitem(1,'user_id',idw_up1.object.user_id[1])
	dw_update.setitem(1,'curr_id',idw_up1.object.curr_id[1])
	dw_update.setitem(1,'kurs',idw_up1.object.kurs[1])
	
	// mencari tgl_due
	ldt_tgl_due = Datetime(RelativeDate(date(ldt_tgl), ll_top_day))
	dw_update.setitem(1,'tgl_due',ldt_tgl_due)
	
	dw_update.accepttext()
	idw_up1.accepttext()
	if idw_up1.update() = 1 and dw_update.update() = 1 then
		commit;
	else
		rollback;
	end if
else
	if idw_up1.update() = 1 then
		commit;
	else
		rollback;
	end if
end if

if idw_down2.rowcount() >0 then
	ls_order_client = this.object.order_client[1]
	if gi_auto_pogl>0 then
		setpointer(hourglass!)
		f_transfer_nonitempo(ls_order_client)
		setpointer(arrow!)

	this.triggerevent('ue_insert')

end event
