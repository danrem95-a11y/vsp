public subroutine wf_neraca ();long							i, ll_row, ll_insert, ll_baris, j
string							ls_desc, ls_account, ls_trd123, ls_t1, ls_t2, ls_t3
double 						ld_cm1=0, ld_cm2=0,ld_cm3=0, ld_ytd1=0, ld_ytd2=0, ld_ytd3=0
double						ld_saldo_cm, ld_saldo_ytd
datetime						ld_period
string							ls_rep_id,ls_sdesc,ls_nrep
double						ld_totcur=0,ld_totytd=0,ld_ttl_mutasi=0,ld_ttl_ytd=0

long							i1, ll_row1, ll_insert1, ll_baris1, j1
string							ls_desc1, ls_account1, ls_trd1231, ls_t11, ls_t21, ls_t31
double 						ld_cm11=0, ld_cm21=0,ld_cm31=0, ld_ytd11=0, ld_ytd21=0, ld_ytd31=0
double						ld_saldo_cm1, ld_saldo_ytd1
datetime						ld_period1
string							ls_site1,ls_rep_id1,ls_sdesc1,ls_nrep1
double						ld_totcur1=0,ld_totytd1=0,ld_ttl_mutasi1=0,ld_ttl_ytd1=0

double						ld_sales1,ld_sales2,ld_sales3,ld_persen
string							ls_sitename, ls_month, ls_year, ls_company,ls_periode
integer 						ll_insert_t,li_tipe_t = 0
string 						ls_trd
datastore				   	lds_report,lds_saldo1,lds_saldo2

setpointer(Hourglass!)
open(w_wait)

SELECT gl_site.company_site  INTO :ls_sitename 
  FROM gl_site  
 WHERE gl_site.site_id = :gs_site   ;
if isnull(ls_sitename) then ls_sitename = 'Demo Version'

if ii_bulan = 1 then
	ii_bulan_before  = 12
	ii_tahun_before  = ii_tahun - 1
else
	ii_bulan_before  = ii_bulan - 1
	ii_tahun_before  = ii_tahun
end if

// retrieve report format
lds_report							= create datastore
lds_report.dataobject 			= 'ds_report_form'
lds_report.settransobject(sqlca)
lds_report.retrieve(gs_site,is_report) 

dw_2.reset()
ls_periode 							= 'PERIODE : ' + upper(f_bulan(ii_bulan,'L')) + ' ' + string(ii_tahun)
ld_period 							= datetime(date(ii_tahun, ii_bulan, 1))

// retrieve neraca percobaan
lds_saldo1 = create datastore
lds_saldo1.dataobject = 'ds_coa_saldo'
lds_saldo1.SetTransObject(sqlca)

lds_saldo2 = create datastore
lds_saldo2.dataobject = 'ds_coa_saldo'
lds_saldo2.SetTransObject(sqlca)

for i = 1 to lds_report.rowcount()
//	if cbx_1.checked = true then
		ls_desc 		= lds_report.getitemstring(i,'description')
//	else
//		ls_desc 		= lds_report.getitemstring(i,'description2')
//	end if
	ls_account	= lds_report.getitemstring(i,'account_id')
	ls_trd123 	= lds_report.getitemstring(i,'trd123')
	ls_t1 			= lds_report.getitemstring(i,'t1')
	ls_t2 			= lds_report.getitemstring(i,'t2')
	ls_t3 			= lds_report.getitemstring(i,'t3')

	choose case ls_trd123
		case 'S' 
			if li_tipe_t = 0 then
				ll_insert =	dw_2.insertrow(0)
				dw_2.object.description[ll_insert] = ls_desc
			else 
				// nilainya digabung dengan nilai yang tipe = T
				ll_baris  = lds_saldo2.Retrieve(ii_bulan,ii_tahun,gs_site,ls_account + '%')
				ll_baris1 = lds_saldo1.Retrieve(ii_bulan_before,ii_tahun_before,gs_site,ls_account + '%')
				
				// untuk nilai
			   if ll_baris > 0 then
					ld_ttl_ytd    = lds_saldo2.getitemnumber(1,'ttl_akhir')
				else
					ld_ttl_ytd    = 0
				end if				
					
				if ls_t1 = '+' then
					ld_totytd = ld_totytd + ld_ttl_ytd
				else
					ld_totytd = ld_totytd - ld_ttl_ytd					
				end if
				
				// untuk nilai2
			   if ll_baris1 > 0 then
					ld_ttl_ytd1    = lds_saldo1.getitemnumber(1,'ttl_akhir')
				else
					ld_ttl_ytd1    = 0
				end if				
					
				if ls_t1 = '+' then
					ld_totytd1 = ld_totytd1 + ld_ttl_ytd1
				else
					ld_totytd1 = ld_totytd1 - ld_ttl_ytd1					
				end if				
			end if	
		case 'T'
			// get data from gl_history and gl_beginning_balance, and then check to gl_coa
			ll_baris  = lds_saldo2.Retrieve(ii_bulan,ii_tahun,gs_site,ls_account + '%')
			ll_baris1 = lds_saldo1.Retrieve(ii_bulan_before,ii_tahun_before,gs_site,ls_account + '%')
			
			if ll_baris > 0 then
				ld_ttl_ytd    = lds_saldo2.getitemnumber(1,'ttl_akhir')
			else
				ld_ttl_ytd    = 0
			end if
				
			if ls_t1 = '+' then
				ld_totytd = ld_totytd + ld_ttl_ytd
			else
				ld_totytd = ld_totytd - ld_ttl_ytd
			end if
			
			if ll_baris1 > 0 then
				ld_ttl_ytd1    = lds_saldo1.getitemnumber(1,'ttl_akhir')
			else
				ld_ttl_ytd1    = 0
			end if
				
			if ls_t1 = '+' then
				ld_totytd1 = ld_totytd1 + ld_ttl_ytd1
			else
				ld_totytd1 = ld_totytd1 - ld_ttl_ytd1
			end if

			li_tipe_t = 1	
			
			ll_insert = dw_2.insertrow(0)
			dw_2.object.description[ll_insert] 	= ls_desc
			dw_2.object.nilai[ll_insert]			= 0	
			dw_2.object.nilai2[ll_insert]			= 0
			ll_insert_t = ll_insert
		case 'R'
			// get data from gl_history and gl_beginning_balance, and then check to gl_coa
			ll_baris  = lds_saldo2.Retrieve(ii_bulan,ii_tahun,gs_site,ls_account + '%')
			ll_baris1 = lds_saldo1.Retrieve(ii_bulan_before,ii_tahun_before,gs_site,ls_account + '%')
			
			if ll_baris > 0 then
				ld_ttl_ytd    = lds_saldo2.getitemnumber(1,'ttl_akhir')
			else
				ld_ttl_ytd    = 0
			end if
		
			ld_saldo_ytd = ld_ttl_ytd
			
			if ll_baris1 > 0 then
				ld_ttl_ytd1    = lds_saldo1.getitemnumber(1,'ttl_akhir')
			else
				ld_ttl_ytd1    = 0
			end if
		
			ld_saldo_ytd1 = ld_ttl_ytd1
			
			ll_insert = dw_2.insertrow(0)
			dw_2.object.description[ll_insert] 	= ls_desc
			if ls_t1 = '+' then
				dw_2.object.nilai[ll_insert] 			= ld_saldo_ytd	
				dw_2.object.nilai2[ll_insert] 			= ld_saldo_ytd1
			else
				dw_2.object.nilai[ll_insert] 			= ld_saldo_ytd	* -1
				dw_2.object.nilai2[ll_insert] 			= ld_saldo_ytd1 * -1
			end if
		case 'D'
			// get data from gl_history and gl_beginning_balance, and then check to gl_coa
			ll_baris  = lds_saldo2.Retrieve(ii_bulan,ii_tahun,gs_site,ls_account + '%')
			ll_baris1 = lds_saldo1.Retrieve(ii_bulan_before,ii_tahun_before,gs_site,ls_account + '%')
			if ll_baris > 0 then
				ld_ttl_ytd    = lds_saldo2.getitemnumber(1,'ttl_akhir')
			else
				ld_ttl_ytd    = 0
			end if
			
			ld_saldo_ytd = ld_ttl_ytd
			
			if ll_baris1 > 0 then
				ld_ttl_ytd1    = lds_saldo1.getitemnumber(1,'ttl_akhir')
			else
				ld_ttl_ytd1    = 0
			end if
			
			ld_saldo_ytd1 = ld_ttl_ytd1
			
			ll_insert = dw_2.insertrow(0)
			dw_2.object.description[ll_insert] 	= ls_desc
			dw_2.object.nilai[ll_insert] 			= ld_saldo_ytd		
			dw_2.object.nilai2[ll_insert] 		   = ld_saldo_ytd1		
		case '1'
			ld_saldo_ytd	= ld_ytd1
			ll_insert 		= dw_2.insertrow(0)
			dw_2.object.description[ll_insert] 	= ls_desc
			dw_2.object.nilai[ll_insert] 			= ld_saldo_ytd	
			
			ld_saldo_ytd1	= ld_ytd11
			dw_2.object.nilai2[ll_insert] 		   = ld_saldo_ytd1
		case '2'
			ld_saldo_ytd 	= ld_ytd2
			ll_insert 		= dw_2.insertrow(0)
			dw_2.object.description[ll_insert] 	= ls_desc
			dw_2.object.nilai[ll_insert] 			= ld_saldo_ytd		
			
			ld_saldo_ytd1 	= ld_ytd21
			dw_2.object.nilai2[ll_insert] 		   =  ld_saldo_ytd1		
		case '3'
			ld_saldo_ytd 	= ld_ytd3
			ll_insert 		= dw_2.insertrow(0)
			dw_2.object.description[ll_insert] 	= ls_desc
			dw_2.object.nilai[ll_insert] 			= ld_saldo_ytd	
			
			ld_saldo_ytd1 	= ld_ytd31
			dw_2.object.nilai2[ll_insert] 		   = ld_saldo_ytd1
	end choose
	
	ls_trd = ''
	if li_tipe_t = 1 then ls_trd = lds_report.getitemstring(i+1,'trd123')
	if li_tipe_t = 1 and (ls_trd <> 'S') then
	   // harus insert 1 row untuk hasil proses T
		dw_2.object.nilai[ll_insert_t]		= ld_totytd
		dw_2.object.nilai2[ll_insert_t]	= ld_totytd1
		
		li_tipe_t = 0
		ls_t1	    = '+'		
		
		ld_saldo_ytd 	= ld_totytd
		ld_saldo_ytd1 	= ld_totytd1
		
		ld_totytd  = 0
		ld_totytd1 = 0
	end if		
	
	// formula
	if ls_t3 = '0' then
		ld_ytd3 = 0
		ld_ytd31 = 0
	end if
	if ls_t3 = '+' then
		ld_ytd3 	= ld_ytd3 + ld_saldo_ytd
		ld_ytd31 = ld_ytd31 + ld_saldo_ytd1
	end if
	if ls_t3 = '-' then
		ld_ytd3 	= ld_ytd3 - ld_saldo_ytd
		ld_ytd31 = ld_ytd31 - ld_saldo_ytd1
	end if
	
	if ls_t2 = '0' then
		ld_ytd2 	= 0
		ld_ytd21 = 0
	end if
	if ls_t2 = '+' then
		ld_ytd2 	= ld_ytd2 + ld_saldo_ytd
		ld_ytd21 = ld_ytd21 + ld_saldo_ytd1
	end if
	if ls_t2 = '-' then
		ld_ytd2 	= ld_ytd2 - ld_saldo_ytd
		ld_ytd21 = ld_ytd21 - ld_saldo_ytd1
	end if

	if ls_t1 = '0' then
		ld_ytd1 	= 0
		ld_ytd11 = 0
	end if
	if ls_t1 = '+' then
		ld_ytd1 	= ld_ytd1 + ld_saldo_ytd
		ld_ytd11 	= ld_ytd11 + ld_saldo_ytd1
	end if
	if ls_t1 = '-' then
		ld_ytd1 	= ld_ytd1 - ld_saldo_ytd
		ld_ytd11 	= ld_ytd11 - ld_saldo_ytd1
	end if
	ld_saldo_ytd = 0
	ld_saldo_ytd1 = 0
	
	//beep(1)
	
next

for i = 1 to dw_2.rowcount()
	dw_2.setitem(i,'company_name',f_company())
//	dw_2.setitem(i,'site_name',ls_sitename)
	dw_2.setitem(i,'periode_name',ls_periode)
next

setpointer(Arrow!)
dw_2.setredraw(true)
long li_flag
li_flag = dw_1.object.flag_zero[1]
if li_flag = 0 then
	dw_2.SetFilter("#3>"+ String(0) + " or #4>" + String(0))
else
	dw_2.setfilter('')
end if
dw_2.filter()
close(w_wait)

end subroutine

event ue_xls;datetime ldt1,ldt2,ldt3,ldt4
long ll_report,ll_jenis,ll_bank
string ls_awal
long ll_year,ll_year2

if dw_2.dataobject <> 'dw_rpt_is_compo' and dw_2.dataobject <> 'dw_rpt_is_compo_rekap'  then
	gurningsoft_xls(dw_2)
	return
end if
dw_1.accepttext()



ldt1 = dw_1.object.tgl1[1]
ll_report = dw_1.object.flag_zero[1]
ldt1 = datetime(date(ldt1))


ldt1 = f_bom(ldt1)
ldt2 = f_eom(ldt1)

ll_year = year(date(ldt1))
ll_year2 = year(date(ldt2))
if ll_year <> ll_year2 then
	messagebox('Warning!','Tidak diperbolehkan beda tahun..!',stopsign!)
	return
end if

ls_awal = string(ll_year)+'-01-01'
ldt3 = datetime(ls_awal)

select dateadd(day,-1,date(:ldt1)) 
into :ldt4
from gl_setup;


dw_1.setitem(1,'tgl1',ldt1)


open(w_wait)
if dw_2.dataobject = 'dw_rpt_is_compo' or dw_2.dataobject = 'dw_rpt_is_compo_rekap' then
	//Get Summary by Group
	decimal ldec_jual_lalu,ldec_jual_ini,ldec_hpp_lalu,ldec_hpp_ini,ldec_biaya_lalu,ldec_biaya_ini
	decimal ldec_dapat_lalu,ldec_dapat_ini,ldec_blain_lalu,ldec_blain_ini
	decimal ldec_pajak_lalu,ldec_pajak_ini
	long i,ll_urut
	dw_rl.retrieve(ldt3,ldt4,ldt1,ldt2)
	dw_rl.accepttext()
	if dw_rl.getrow() <=0 then return
	ldec_jual_lalu = dw_rl.object.penjualan_lalu[1]
	ldec_jual_ini = dw_rl.object.penjualan_ini[1]
	ldec_hpp_lalu = dw_rl.object.hpp_lalu[1]
	ldec_hpp_ini = dw_rl.object.hpp_ini[1]
	ldec_biaya_lalu = dw_rl.object.biaya_lalu[1]
	ldec_biaya_ini = dw_rl.object.biaya_ini[1]
	ldec_dapat_lalu = dw_rl.object.pendapatan_lalu[1]
	ldec_dapat_ini = dw_rl.object.pendapatan_ini[1]
	ldec_blain_lalu = dw_rl.object.biaya_lain_lalu[1]
	ldec_blain_ini = dw_rl.object.biaya_lain_ini[1]
	ldec_pajak_lalu = dw_rl.object.pajak_lalu[1]
	ldec_pajak_ini = dw_rl.object.pajak_ini[1]
	
	dw_update.reset()
	ll_urut = 0
	dw_rl1.retrieve(ldt3,ldt4,ldt1,ldt2,ldec_jual_lalu,ldec_jual_ini,ldec_hpp_lalu,ldec_hpp_ini,ldec_biaya_lalu,ldec_biaya_ini,ldec_dapat_lalu,ldec_dapat_ini,ldec_blain_lalu,ldec_blain_ini,1)
	dw_rl1.accepttext()
	if dw_rl1.rowcount() > 0 then
		for i = 1 to dw_rl1.rowcount()
			ll_urut ++
			dw_update.setrow(dw_update.insertrow(0))
			dw_update.setitem(dw_update.getrow(),'parent_urut',1)
			dw_update.setitem(dw_update.getrow(),'user_id',gs_user)
			dw_update.setitem(dw_update.getrow(),'tgl1',ldt1)
			dw_update.setitem(dw_update.getrow(),'tgl2',ldt2)
			dw_update.setitem(dw_update.getrow(),'dataobject',dw_rl1.dataobject)
			dw_update.setitem(dw_update.getrow(),'fincatcode',dw_rl1.object.fincatcode[i])
			dw_update.setitem(dw_update.getrow(),'fincatdes',dw_rl1.object.fincatdes[i])
			dw_update.setitem(dw_update.getrow(),'parentcode',dw_rl1.object.parentcode[i])
			dw_update.setitem(dw_update.getrow(),'parentdes',dw_rl1.object.parentname[i])
			dw_update.setitem(dw_update.getrow(),'accountcode',dw_rl1.object.accountcode[i])
			dw_update.setitem(dw_update.getrow(),'accountdes',dw_rl1.object.accountdes[i])
			dw_update.setitem(dw_update.getrow(),'lalu',dw_rl1.object.clalu[i])
			dw_update.setitem(dw_update.getrow(),'ini',dw_rl1.object.cmutasi[i])
			dw_update.setitem(dw_update.getrow(),'sdini',dw_rl1.object.cini[i])
			dw_update.setitem(dw_update.getrow(),'jual_lalu',ldec_jual_lalu)
			dw_update.setitem(dw_update.getrow(),'jual_ini',ldec_jual_ini)
			dw_update.setitem(dw_update.getrow(),'hpp_lalu',ldec_hpp_lalu)
			dw_update.setitem(dw_update.getrow(),'hpp_ini',ldec_hpp_ini)
			dw_update.setitem(dw_update.getrow(),'biaya_lalu',ldec_biaya_lalu)
			dw_update.setitem(dw_update.getrow(),'biaya_ini',ldec_biaya_ini)
			dw_update.setitem(dw_update.getrow(),'dapat_lalu',ldec_dapat_lalu)
			dw_update.setitem(dw_update.getrow(),'dapat_ini',ldec_dapat_ini)
			dw_update.setitem(dw_update.getrow(),'biaya_lain_lalu',ldec_blain_lalu)
			dw_update.setitem(dw_update.getrow(),'biaya_lain_ini',ldec_blain_ini)
			dw_update.setitem(dw_update.getrow(),'pajak_lalu',ldec_pajak_lalu)
			dw_update.setitem(dw_update.getrow(),'pajak_ini',ldec_pajak_ini)
			dw_update.setitem(dw_update.getrow(),'urut',dw_update.getrow())
		next
	end if
	//RL 2
	dw_rl2.retrieve(ldt3,ldt4,ldt1,ldt2,ldec_jual_lalu,ldec_jual_ini,ldec_hpp_lalu,ldec_hpp_ini,ldec_biaya_lalu,ldec_biaya_ini,ldec_dapat_lalu,ldec_dapat_ini,ldec_blain_lalu,ldec_blain_ini,1)
	dw_rl2.accepttext()
	if dw_rl2.rowcount() > 0 then
		for i = 1 to dw_rl2.rowcount()
			ll_urut ++
			dw_update.setrow(dw_update.insertrow(0))
			dw_update.setitem(dw_update.getrow(),'parent_urut',2)
